#!/bin/sh
echo "enter file name"
read file
if [-d $file];
 then
   echo "file is the directory"
 elif [-f $file];
   then 
     if [-r $file -a -w $file -a -x $file ];
       then 
         echo "echo you have read,write,execute access in the file"
     else
        echo "this is the ordinary file"
     fi
  else
       echo "this is neither file nor directory"
fi
